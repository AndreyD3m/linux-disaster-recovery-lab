## Использованные команды 
```bash
df -h # проверить текущее состояние диска
dd if=/dev/zero of=~/bigfile bs=1G count=15 # создаёт файл большого размера
du -ah ~/ | sort -rh | head -n 10 # показывает 10 самых тяжелых файлов в домашней директории
rm -f ~/bigfile # удалить большой файл
lsof | grep deleted # поиск открытых удаленных файлов
sudo systemctl restart имя_сервиса # перезапуск процесса 
sudo kill -9 PID # убийство процесса
